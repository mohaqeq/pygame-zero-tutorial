def erstelle_partikel(x, y, farbe):
    import random
    for i in range(10):
        p = {
            "x": x,
            "y": y,
            "speed_x": random.uniform(-5, 5),
            "speed_y": random.uniform(-5, 5),

        partikel.append(p)

def update_partikel():
    for p in partikel[:]:
        p["x"] = p["x"] + p["speed_x"]
        p["y"] = p["y"] + p["speed_y"]
        p["size"] = p["size"] * 0.95  # Wird kleiner
        p["leben"] = p["leben"] - 1
        if p["leben"] <= 0:
            partikel.remove(p)

def draw_partikel():
    for p in partikel:
        screen.draw.filled_circle(
            (int(p["x"]), int(p["y"])),
            int(p["size"]),
            p["farbe"]
        )

def speichere_highscore(punkte):
    try:
